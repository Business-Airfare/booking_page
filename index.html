// supabase/functions/track_payment_link_event/index.ts import { serve } from "https://deno.land/std@0.224.0/http/server.ts"; import { createClient } from "npm:@supabase/supabase-js@2"; import { paymentPageCors } from "../_shared/payment_page_cors.ts"; import { requirePaymentSessionRowExists } from "../_shared/payment_session_gate.ts"; function cors(origin: string | null) { return paymentPageCors(origin, { allowMethods: "POST, OPTIONS", fallbackOrigin: "https://booking.business-airfare.com", }); } function json(origin: string | null, status: number, body: unknown) { return new Response(JSON.stringify(body), { status, headers: { ...cors(origin), "content-type": "application/json; charset=utf-8", "cache-control": "no-store", }, }); } function extractIp(req: Request): string | null { const forwardedFor = req.headers.get("x-forwarded-for"); if (forwardedFor) { const firstIp = forwardedFor.split(",")[0]?.trim(); if (firstIp) return firstIp; } const cfIp = req.headers.get("cf-connecting-ip"); if (cfIp) return cfIp.trim(); return null; } function fullName(first?: string | null, middle?: string | null, last?: string | null) { return [first, middle, last].filter((v) => typeof v === "string" && v.trim()).join(" ").trim(); } function escapeLike(s: string) { // For LIKE patterns: escape %, _ and backslash return s.replace(/\\/g, "\\\\").replace(/%/g, "\\%").replace(/_/g, "\\_"); } serve(async (req) => { const origin = req.headers.get("origin"); if (req.method === "OPTIONS") { return new Response(null, { status: 200, headers: cors(origin) }); } if (req.method !== "POST") { return json(origin, 405, { error: "Method not allowed" }); } try { // This edge function is deployed in the payment project, but we may need to write to the CRM project. // Configure these secrets on the edge function if your payment + CRM are on different Supabase projects. const crmSupabaseUrl = Deno.env.get("CRM_SUPABASE_URL") ?? Deno.env.get("SUPABASE_URL"); const crmServiceKey = Deno.env.get("CRM_SUPABASE_SERVICE_ROLE_KEY") ?? Deno.env.get("SUPABASE_SERVICE_ROLE_KEY"); if (!crmSupabaseUrl || !crmServiceKey) { return json(origin, 500, { error: "Missing CRM_SUPABASE_URL / CRM_SUPABASE_SERVICE_ROLE_KEY secrets" }); } const db = createClient(crmSupabaseUrl, crmServiceKey, { auth: { persistSession: false } }); let body: any; try { body = await req.json(); } catch { return json(origin, 400, { error: "Invalid JSON body" }); } const tracking_token = String(body?.tracking_token ?? "").trim(); if (!tracking_token) return json(origin, 400, { error: "tracking_token is required" }); const event_type = String(body?.event_type ?? "").trim(); if (!event_type) return json(origin, 400, { error: "event_type is required" }); // In this MVP we map the payment session public_id (payment link id) → price_quote → request_id. const payment_link_id = body?.payment_link_id == null ? null : String(body.payment_link_id).trim(); if (!payment_link_id) return json(origin, 400, { error: "payment_link_id is required for request resolution" }); // Use row-exists gate (not active-checkout only): same as get_payment_link_chat — we still want // analytics when the link is opened after payment or after expiry, and public_id alone is not secret. const sessionGate = await requirePaymentSessionRowExists(payment_link_id); if (!sessionGate.ok) { return json(origin, sessionGate.status, { error: sessionGate.error }); } const metadataInput = body?.metadata ?? {}; const metadata = metadataInput && typeof metadataInput === "object" && !Array.isArray(metadataInput) ? metadataInput : {}; const ip = extractIp(req); const user_agent = req.headers.get("user-agent"); // 1) Resolve request_id from payment_link_id const pattern = `%${escapeLike(payment_link_id)}%`; const { data: pqRows, error: pqErr } = await db .from("price_quotes") .select("id, request_id, status, created_at, booking_url, created_by") .ilike("booking_url", pattern); if (pqErr) return json(origin, 500, { error: "DB error resolving payment link", details: pqErr.message }); if (!pqRows || pqRows.length === 0) { return json(origin, 404, { error: "No request found for this payment_link_id" }); } // Prefer sent, then sold; otherwise latest row. const sorted = [...pqRows].sort((a: any, b: any) => new Date(b.created_at ?? 0).getTime() - new Date(a.created_at ?? 0).getTime()); const sent = sorted.find((pq: any) => String(pq.status ?? "").toLowerCase() === "sent"); const sold = sorted.find((pq: any) => String(pq.status ?? "").toLowerCase() === "sold"); const targetPq = sent ?? sold ?? sorted[0]; const requestId = targetPq.request_id as string | undefined; if (!requestId) return json(origin, 404, { error: "Resolved PQ has no request_id" }); // 2) Enforce 15+ minutes apart (skip if too soon) const TOO_SOON_MS = 15 * 60 * 1000; const { data: lastRow, error: lastErr } = await db .from("payment_link_events") .select("created_at") .eq("tracking_token", tracking_token) .eq("event_type", event_type) .order("created_at", { ascending: false }) .limit(1) .maybeSingle(); if (lastErr) { // Don't fail tracking because of history read issues; just proceed with insert. console.warn("[track_payment_link_event] last event read error:", lastErr.message); } else if (lastRow?.created_at) { const lastTs = new Date(lastRow.created_at).getTime(); if (Number.isFinite(lastTs) && Date.now() - lastTs < TOO_SOON_MS) { return json(origin, 200, { ok: true, inserted: false, skipped: true, reason: "too_soon" }); } } // 3) Insert event (unique index will also prevent duplicates) const insertPayload = { request_id: requestId, payment_link_id, tracking_token, event_type, ip, user_agent, metadata, }; const { data: inserted, error: insErr } = await db .from("payment_link_events") .insert(insertPayload) .select("id, created_at") .single(); if (insErr) { // Handle constraint violations gracefully (unique/exclusion constraints). const code = String((insErr as any)?.code ?? ""); const msg = String((insErr as any)?.message ?? "").toLowerCase(); if ( code === "23505" || // unique_violation code === "23P01" || // exclusion_violation (commonly) msg.includes("duplicate") || msg.includes("rate limit") || msg.includes("payment_link_events: duplicate within 15 minutes") || msg.includes("exclusion constraint") || msg.includes("rate_limit_15min") ) { return json(origin, 200, { ok: true, inserted: false, skipped: true, reason: "duplicate" }); } return json(origin, 500, { error: "DB error inserting payment_link_event", details: insErr.message }); } const insertedId = inserted?.id; // 4) Create CRM notification on first/allowed events (any primary_queue, including closed) const notifyTypes: Record string; body: (requestLabel: string) => string }> = { opened: { notifType: "payment_link_opened", title: (clientName) => `Payment link opened: ${clientName}`, body: (requestLabel) => `Payment link was opened for request ${requestLabel}.`, }, checkout_started: { notifType: "payment_link_checkout_started", title: (clientName) => `Payment checkout started: ${clientName}`, body: (requestLabel) => `Customer started checkout (Book) for request ${requestLabel}.`, }, }; const notifySpec = notifyTypes[event_type]; if (notifySpec) { try { const { data: reqRow, error: reqErr } = await db .from("requests") .select("id, org_id, request_owner_id, request_code, client_id") .eq("id", requestId) .maybeSingle(); if (!reqErr && reqRow?.org_id) { const requestLabel = (reqRow.request_code as string | null) ?? String(reqRow.id); const requestLink = `/requests/${requestLabel}`; let clientRow: { first_name?: string | null; middle_name?: string | null; last_name?: string | null; } | null = null; if (reqRow.client_id) { const { data: c } = await db .from("clients") .select("first_name, middle_name, last_name") .eq("id", reqRow.client_id) .maybeSingle(); clientRow = c ?? null; } const clientName = fullName(clientRow?.first_name, clientRow?.middle_name, clientRow?.last_name) || "—"; const recipientId = (reqRow.request_owner_id as string | null) ?? ((targetPq as { created_by?: string | null }).created_by as string | null) ?? null; await db.from("notifications").insert({ org_id: reqRow.org_id, recipient_id: recipientId, type: notifySpec.notifType, title: notifySpec.title(clientName), body: notifySpec.body(requestLabel), link: requestLink, request_id: requestId, metadata: { payment_link_id, tracking_token, payment_link_event_id: insertedId, }, }); } } catch (notifyErr) { // Best-effort: do not fail tracking if notification fails. console.warn("[track_payment_link_event] notification insert error:", (notifyErr as any)?.message ?? notifyErr); } } return json(origin, 200, { ok: true, inserted: true, id: insertedId, created_at: inserted?.created_at }); } catch (e) { return json(origin, 500, { error: "Server error", details: String(e) }); } });
